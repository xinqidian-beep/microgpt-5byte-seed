2. 保存/恢复完整伪代码
def save_state(self, filepath: str, incremental: bool = False):
        """完整/增量保存框架"""
        if incremental and self.save_counter % 5 != 0:  # 每5次完整保存才做一次增量（可调）
            self.save_counter += 1
            return

        state = {
            "version": self.version,
            "step": self.step,
            "current_rhythm_core": int(self.current_rhythm_core),
            "rhythm_history": list(self.rhythm_history),   # 32个整数
            "metadata": self.metadata,
            # 每层核心状态（只保存必要数据，极致压缩）
            "layers": [
                {
                    "layer_id": i+1,
                    "activations_last_100": layer.get_recent_activations(100),  # 仅最近100步摘要
                    "internal_state": layer.serialize_minimal()               # 层自定义极简序列化
                }
                for i, layer in enumerate(self.layers)
            ]
        }

        # 序列化 + 压缩
        json_str = json.dumps(state, separators=(',', ':'))  # 极致紧凑
        compressed = zlib.compress(json_str.encode('utf-8'), level=9)

        with open(filepath, "wb") as f:
            f.write(compressed)

        self.metadata["last_save_step"] = self.step
        self.save_counter += 1

        print(f"✅ 保存完成: {filepath} ({'增量' if incremental else '完整'}，{len(compressed)/1024:.1f} KB)")

    def load_state(self, filepath: str):
        """断点恢复 + 版本兼容"""
        with open(filepath, "rb") as f:
            compressed = f.read()
        
        json_str = zlib.decompress(compressed).decode('utf-8')
        state = json.loads(json_str)

        # ==================== 版本兼容性处理 ====================
        if state["version"] != self.version:
            print(f"⚠️ 版本不匹配: 存档 {state['version']}，当前 {self.version}")
            # 自动迁移逻辑（未来可扩展）
            if state["version"] in ["v3.0", "v3.5"]:
                print("→ 执行v3→v4迁移：RhythmCore历史自动补零")
                state.setdefault("rhythm_history", [0] * 32)
            else:
                raise ValueError("不支持的旧版本存档，请使用v4.0+存档")

        # ==================== 恢复核心状态 ====================
        self.step = state["step"]
        self.current_rhythm_core = state["current_rhythm_core"]
        self.rhythm_history.clear()
        self.rhythm_history.extend(state["rhythm_history"])
        self.metadata = state["metadata"]
        self.metadata["last_save_step"] = self.step  # 更新

        # 恢复每层状态
        for i, layer_data in enumerate(state["layers"]):
            self.layers[i].deserialize_minimal(layer_data["internal_state"])
            # 可选：用最近激活值做热启动平滑
            self.layers[i].warmup_from_recent(layer_data["activations_last_100"])

        print(f"✅ 恢复完成: 从第 {self.step} 步继续运行（RhythmCore={self.current_rhythm_core}）")
        return True

def auto_save(self, base_path: str = "/data/microbrain/"):
        """手机端后台轻量增量保存（每5~10分钟调用一次）"""
        timestamp = int(time.time())
        filepath = f"{base_path}brain_state_{self.step}.mbak"  # .mbak = microbrain archive

        # 增量模式（只在步数增长显著时保存）
        if self.step - self.metadata.get("last_save_step", 0) > 5000:
            self.save_state(filepath, incremental=True)
            # 可选：保留最近3个存档，自动清理旧文件（手机存储友好）
            self.cleanup_old_saves(base_path, keep=3)

        # ...

    def get_recent_activations(self, n: int = 100) -> list:
        return self.activation_buffer[-n:] if hasattr(self, 'activation_buffer') else []

    def warmup_from_recent(self, recent_acts: list):
        """热启动平滑，避免恢复后突变"""
